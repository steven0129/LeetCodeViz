LeetCode 11 - Container With Most Water

class Solution:


    def maxArea(self, height: List[int]) -> int:


        ans = 0


        left = 0


        right = len(height) - 1


        highest = max(height)



        while left < right:


            area = min(height[right], height[left]) * (right - left)


            ans = max(ans, area)



            if highest * (right - left) < ans:  # 效率提升的關鍵


                break



            if height[right] > height[left]:


                left += 1


            else:


                right -= 1



        return ans
